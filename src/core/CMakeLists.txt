        ${CMAKE_CURRENT_SOURCE_DIR}
        "${PROJECT_SOURCE_DIR}/include/MyScene/core"
)

TO_PACKAGE_NAME(ECS_PACKAGE_NAME MyECS ${MyECS_VERSION})
TO_PACKAGE_NAME(DP_PACKAGE_NAME MyDP ${MyDP_VERSION})
TO_PACKAGE_NAME(TEMPLATE_PACKAGE_NAME MyTemplate ${MyTemplate_VERSION})

        SOURCES ${sources}
        INTERFACE_INC ON
        LIBS
        "My::${GM_PACKAGE_NAME}_core"
        "My::${ECS_PACKAGE_NAME}_core"
        "My::${ECS_PACKAGE_NAME}_cmpt"
        "My::${DP_PACKAGE_NAME}_core"
        "My::${TEMPLATE_PACKAGE_NAME}_core"
)
